<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";
import { getFirestore, doc, setDoc, onSnapshot, getDoc, collection, getDocs, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";

const db = getFirestore(initializeApp({
  apiKey:"AIzaSyDTrYvYgHB57sF_CHV-VJyjFpTsbDfNJoA", authDomain:"instapquiz.firebaseapp.com",
  projectId:"instapquiz", storageBucket:"instapquiz.firebasestorage.app",
  messagingSenderId:"165422140837", appId:"1:165422140837:web:f8541e95c467e306e3ec4c"
}));

let username="", roomCode="", myScore=0, currentQuestionId=null, answered=false;
let timerInterval=null, questionStartTime=null, currentTimeLimit=30, quizUnsub=null;

document.getElementById('room-input').addEventListener('input', e => { e.target.value=e.target.value.replace(/\D/g,''); });

window.joinQuiz = async function() {
  const code=document.getElementById('room-input').value.trim();
  const name=document.getElementById('username-input').value.trim();
  if (code.length!==4) return showError("Geef een geldige 4-cijferige code in.");
  if (!name||name.length<2) return showError("Naam moet minstens 2 tekens zijn.");
  const roomSnap=await getDoc(doc(db,"rooms",code,"quiz","state"));
  if (!roomSnap.exists()) return showError("Room niet gevonden. Controleer de code.");
  if (roomSnap.data().status==='finished') return showError("Deze quiz is al afgelopen.");
  const playerSnap=await getDoc(doc(db,"rooms",code,"players",name));
  if (playerSnap.exists()) return showError("Deze naam is al in gebruik in deze room.");
  username=name; roomCode=code;
  await setDoc(doc(db,"rooms",code,"players",name),{score:0,joinedAt:serverTimestamp()});
  document.getElementById('wait-name').textContent=`Ingelogd als: ${name}`;
  document.getElementById('wait-room').textContent=`Room ${code}`;
  document.getElementById('page-subtitle').textContent=`Room ${code}`;
  showScreen('waiting');
  startListening();
};

function showError(msg){const el=document.getElementById('login-error');el.textContent=msg;el.style.display='block';}
function showScreen(name){
  ['login','waiting','question','leaderboard','finished','abandoned'].forEach(s=>document.getElementById(`screen-${s}`).classList.add('hidden'));
  document.getElementById(`screen-${name}`).classList.remove('hidden');
}

function startListening(){
  if(quizUnsub)quizUnsub();
  quizUnsub=onSnapshot(doc(db,"rooms",roomCode,"quiz","state"),async snap=>{
    if(!snap.exists())return;
    const s=snap.data();
    if(s.status==="waiting")showScreen('waiting');
    else if(s.status==="question"&&s.questionId!==currentQuestionId){currentQuestionId=s.questionId;answered=false;await loadQuestion(s.questionId,s.timeLimit||30,s.questionStartedAt);}
    else if(s.status==="leaderboard"){clearInterval(timerInterval);await showLeaderboard();}
    else if(s.status==="finished"){clearInterval(timerInterval);await showFinished();}
    else if(s.status==="abandoned"){clearInterval(timerInterval);showAbandoned();}
  });
}

async function loadQuestion(qId,timeLimit,startedAt){
  const qSnap=await getDoc(doc(db,"rooms",roomCode,"questions",qId));
  if(!qSnap.exists())return;
  const q=qSnap.data();
  currentTimeLimit=timeLimit;
  showScreen('question');
  document.getElementById('q-number').textContent=`Vraag ${q.order||''}`;
  document.getElementById('question-text').textContent=q.question;
  document.getElementById('feedback-bar').className='feedback-bar';
  const area=document.getElementById('answer-area');
  area.innerHTML='';
  if(q.type==='multiple'){
    const opts=document.createElement('div');opts.className='options';
    q.options.forEach((opt,i)=>{
      const btn=document.createElement('button');btn.className='option-btn';
      btn.innerHTML=`<span class="opt-label">${['A','B','C','D'][i]}</span>${opt}`;
      btn.onclick=()=>submitAnswer(opt,q,btn,opts);opts.appendChild(btn);
    });area.appendChild(opts);
  }else if(q.type==='truefalse'){
    const opts=document.createElement('div');opts.className='options';
    ['Juist','Fout'].forEach(val=>{
      const btn=document.createElement('button');btn.className='option-btn';
      btn.innerHTML=`<span class="opt-label">${val==='Juist'?'✓':'✗'}</span>${val}`;
      btn.onclick=()=>submitAnswer(val,q,btn,opts);opts.appendChild(btn);
    });area.appendChild(opts);
  }else if(q.type==='short'){
    const wrap=document.createElement('div');wrap.className='open-answer';
    const inp=document.createElement('input');inp.type='text';inp.placeholder='Jouw antwoord...';inp.autocomplete='off';
    inp.style.cssText='width:100%;background:#ffffff0a;border:2px solid #ffffff18;border-radius:10px;color:var(--text);font-family:DM Sans,sans-serif;font-size:1.1rem;padding:14px 16px;outline:none;margin-bottom:12px;';
    inp.addEventListener('keydown',e=>{if(e.key==='Enter')sbtn.click();});
    const sbtn=document.createElement('button');sbtn.className='btn';sbtn.textContent='Indienen →';
    sbtn.onclick=()=>submitAnswer(inp.value.trim(),q,sbtn,wrap);
    wrap.appendChild(inp);wrap.appendChild(sbtn);area.appendChild(wrap);
  }else{
    const wrap=document.createElement('div');wrap.className='open-answer';
    const ta=document.createElement('textarea');ta.placeholder='Jouw antwoord...';ta.rows=3;
    const sbtn=document.createElement('button');sbtn.className='btn';sbtn.textContent='Indienen →';
    sbtn.onclick=()=>submitAnswer(ta.value.trim(),q,sbtn,wrap);
    wrap.appendChild(ta);wrap.appendChild(sbtn);area.appendChild(wrap);
  }
  clearInterval(timerInterval);
  const elapsed=startedAt?(Date.now()-startedAt.toMillis())/1000:0;
  let remaining=Math.max(0,timeLimit-elapsed);
  questionStartTime=Date.now()-elapsed*1000;
  updateTimer(remaining,timeLimit);
  timerInterval=setInterval(()=>{
    remaining-=1;
    if(remaining<=0){clearInterval(timerInterval);if(!answered){disableAnswers();showFeedback('time');}remaining=0;}
    updateTimer(remaining,timeLimit);
  },1000);
}

function updateTimer(r,total){
  document.getElementById('timer-circle').style.strokeDashoffset=126*(1-r/total);
  document.getElementById('timer-circle').style.stroke=r>10?'var(--accent)':'var(--accent2)';
  document.getElementById('timer-label').textContent=Math.ceil(r);
}

async function submitAnswer(value,question,btn,container){
  if(answered||!value)return;
  answered=true;clearInterval(timerInterval);
  const elapsed=(Date.now()-questionStartTime)/1000;
  const isCorrect=checkAnswer(value,question);
  const points=isCorrect?Math.round(100+Math.max(0,currentTimeLimit-elapsed)*10):0;
  myScore+=points;
  await setDoc(doc(db,"rooms",roomCode,"answers",`${currentQuestionId}_${username}`),{username,questionId:currentQuestionId,answer:value,correct:isCorrect,points,elapsed:Math.round(elapsed*10)/10,timestamp:serverTimestamp()});
  await setDoc(doc(db,"rooms",roomCode,"players",username),{score:myScore},{merge:true});
  disableAnswers();
  if(question.type!=='open'&&question.type!=='short')highlightAnswer(value,question,container);
  showFeedback(isCorrect?'correct':'wrong',points,question.type==='open',question);
}

function checkAnswer(v,q){if(q.type==='open')return true;return v.toLowerCase().trim()===q.answer.toLowerCase().trim();}
function disableAnswers(){document.querySelectorAll('.option-btn,.open-answer .btn,.open-answer textarea').forEach(el=>el.disabled=true);}
function highlightAnswer(value,question,container){
  container.querySelectorAll('.option-btn').forEach(btn=>{
    const text=btn.textContent.trim().slice(1).trim();
    if(text===value)btn.classList.add(checkAnswer(value,question)?'correct':'wrong');
    if(text===question.answer&&!checkAnswer(value,question))btn.classList.add('correct');
  });
}
function showFeedback(type,points,isOpen,q){
  const bar=document.getElementById('feedback-bar');
  if(isOpen&&q&&q.type==='open'){bar.className='feedback-bar open';bar.textContent='✓ Antwoord ingediend!';return;}
  if(type==='correct'){bar.className='feedback-bar correct';bar.textContent=`✓ Juist! +${points} punten`;}
  else if(type==='wrong'){bar.className='feedback-bar wrong';bar.textContent='✗ Fout. +0 punten';}
  else{bar.className='feedback-bar wrong';bar.textContent='⏰ Tijd voorbij!';}
}

async function showLeaderboard(){
  showScreen('leaderboard');
  const snap=await getDocs(collection(db,"rooms",roomCode,"players"));
  const players=[];snap.forEach(d=>players.push({name:d.id,...d.data()}));
  players.sort((a,b)=>b.score-a.score);
  const medals=['🥇','🥈','🥉'];
  const top3=players.slice(0,3);
  // Top 3
  document.getElementById('lb-top3').innerHTML=top3.map((p,i)=>`
    <div class="lb-row">
      <span class="lb-rank top">${medals[i]}</span>
      <span class="lb-name ${p.name===username?'me':''}">${p.name}${p.name===username?' (jij)':''}</span>
      <span class="lb-score">${p.score} pt</span>
    </div>`).join('');
  // Own position
  const myRank=players.findIndex(p=>p.name===username)+1;
  const myScore=players.find(p=>p.name===username)?.score||0;
  if(myRank>3){
    document.getElementById('lb-my-position').innerHTML=`
      <div style="text-align:center;padding:10px 0;">
        <div style="color:var(--muted);font-size:0.82rem;margin-bottom:4px;">Jouw positie</div>
        <div style="font-family:Syne,sans-serif;font-size:2rem;font-weight:800;color:var(--accent);">#${myRank}</div>
        <div style="color:var(--muted);font-size:0.85rem;">${myScore} pt</div>
      </div>`;
  } else {
    document.getElementById('lb-my-position').innerHTML='';
  }
}

async function showFinished(){
  showScreen('finished');
  const snap=await getDocs(collection(db,"rooms",roomCode,"players"));
  const players=[];snap.forEach(d=>players.push({name:d.id,...d.data()}));
  players.sort((a,b)=>b.score-a.score);
  const medals=['🥇','🥈','🥉'];
  const top3=players.slice(0,3);
  const myRank=players.findIndex(p=>p.name===username)+1;
  const myScoreFinal=players.find(p=>p.name===username)?.score||0;
  // Top 3
  document.getElementById('final-top3').innerHTML=
    '<div style="font-family:Syne,sans-serif;font-size:0.82rem;color:var(--muted);text-align:center;margin-bottom:12px;text-transform:uppercase;letter-spacing:1px;">Top 3</div>'+
    top3.map((p,i)=>`
      <div class="lb-row">
        <span class="lb-rank top">${medals[i]}</span>
        <span class="lb-name ${p.name===username?'me':''}">${p.name}${p.name===username?' (jij)':''}</span>
        <span class="lb-score">${p.score} pt</span>
      </div>`).join('');
  // Own position
  const posHtml = myRank <= 3
    ? '<p style="color:var(--green);font-family:Syne,sans-serif;font-weight:700;text-align:center;margin-top:8px;">🎉 Jij staat in de top 3!</p>'
    : `<div style="text-align:center;padding:10px 0;">
        <div style="color:var(--muted);font-size:0.82rem;margin-bottom:4px;">Jouw eindpositie</div>
        <div style="font-family:Syne,sans-serif;font-size:2rem;font-weight:800;color:var(--accent);">#${myRank}</div>
        <div style="color:var(--muted);font-size:0.85rem;">${myScoreFinal} pt · van ${players.length} spelers</div>
      </div>`;
  document.getElementById('final-my-position').innerHTML=posHtml;
}
function showAbandoned(){
  showScreen('abandoned');
}
